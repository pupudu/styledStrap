# Themes

// TODO: Show an input box with theme as a json to allow customizing or pasting a whole new theme
// TODO: Show multiple themes and clicking on them would change the theme
import { setState, getState } from '../core/wrapper';
import { Button } from '../Button';

<Button color="primary" onClick={()=> {
    const theme = getState();
    setState({
      ...theme,
      colors: {
        ...theme.colors,
        primary: `#${Math.round(Math.random() * 10)}${Math.round(Math.random() * 10)}${Math.round(Math.random() * 10)}`,
      }
  });
}}>Change</Button>
